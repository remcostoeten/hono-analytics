# Examples

Quick copy-paste examples to get analytics working in your app.

## Next.js (Recommended)

### Setup in Layout

```tsx path=null start=null
// app/layout.tsx
import { NextJSAnalyticsProvider } from '@hono-analytics/sdk'

export default function RootLayout({ children }) {
  return (
    <html>
      <body>
        <NextJSAnalyticsProvider
          options={{
            apiKey: process.env.NEXT_PUBLIC_ANALYTICS_API_KEY!,
            projectId: process.env.NEXT_PUBLIC_ANALYTICS_PROJECT_ID!,
            excludePatterns: ['/admin/.*', '/api/.*']
          }}
        >
          {children}
        </NextJSAnalyticsProvider>
      </body>
    </html>
  )
}
```

### Track Events

```tsx path=null start=null
// Any component
import { useAnalytics } from '@hono-analytics/sdk'

function ProductPage() {
  const analytics = useAnalytics()
  
  async function handlePurchase() {
    await analytics.track({ url: '/purchase/button-click' })
  }
  
  return <button onClick={handlePurchase}>Buy Now</button>
}
```

## React

```tsx path=null start=null
// App.tsx
import { AnalyticsProvider, useAnalytics } from '@hono-analytics/sdk/react'

function App() {
  return (
    <AnalyticsProvider
      apiKey={process.env.REACT_APP_ANALYTICS_API_KEY!}
      projectId={process.env.REACT_APP_ANALYTICS_PROJECT_ID!}
    >
      <YourApp />
    </AnalyticsProvider>
  )
}

function YourComponent() {
  const analytics = useAnalytics()
  
  const handleClick = () => {
    analytics.track({ url: '/button/click' })
  }
  
  return <button onClick={handleClick}>Click me</button>
}
```

## Vanilla JS

```javascript path=null start=null
import { initAnalytics, track } from '@hono-analytics/sdk'

const analytics = initAnalytics({
  apiKey: 'your-api-key',
  projectId: 'your-project-id'
})

// Track events
track({ url: '/page/view' })
track({ url: '/button/click' })
```

## Vue.js

```javascript path=null start=null
import { initAnalytics, track } from '@hono-analytics/sdk'

const analytics = initAnalytics({
  apiKey: process.env.VUE_APP_ANALYTICS_API_KEY,
  projectId: process.env.VUE_APP_ANALYTICS_PROJECT_ID
})

export default {
  async mounted() {
    await track({ url: this.$route.path })
  },
  
  methods: {
    async trackClick() {
      await track({ url: '/button/click' })
    }
  }
}
```
